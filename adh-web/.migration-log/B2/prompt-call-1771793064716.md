export const extraitApi = {

    ),

    societe: string,
    codeAdherent: number,
    filiation: number,
    dateDebut?: string,
    dateFin?: string,

    apiClient.get<
      ApiResponse<{

    ),
  printExtrait: (data: PrintExtraitRequest) =>
    apiClient.post<ApiResponse<{ jobId: string }>>(
      '/extrait/print',

    ),
